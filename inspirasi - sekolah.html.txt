  <div class="card">
    <h3>💰 Kas Kelas</h3>
    <p>Total kas terkumpul: <strong id="kasTotal">Rp 0</strong></p>
    <input type="number" id="jumlahKas" placeholder="Masukkan jumlah (Rp)" />
    <button onclick="tambahKas()">Bayar Kas</button>
  </div>


  <div class="card">
    <h3>📋 Absensi</h3>
    <input type="text" id="namaSiswa" placeholder="Nama siswa" />
    <button onclick="absen()">Absen Sekarang</button>
    <ul id="daftarAbsensi"></ul>
  </div>


  <div class="card">
    <h3>💡 Inspirasi Kegiatan</h3>
    <input type="text" id="ide" placeholder="Tulis ide kegiatan..." />
    <button onclick="tambahIde()">Kirim Ide</button>
    <ul id="daftarIde"></ul>
  </div>

<script>
  let kas = 0;


  function tambahKas() {
    const jumlah = Number(document.getElementById('jumlahKas').value);
    if (jumlah > 0) {
      kas += jumlah;
      document.getElementById('kasTotal').innerText = `Rp ${kas.toLocaleString('id-ID')}`;
      alert('Kas berhasil ditambahkan!');
      document.getElementById('jumlahKas').value = '';
    } else {
      alert('Masukkan jumlah kas yang valid.');
    }
  }


  function absen() {
    const nama = document.getElementById('namaSiswa').value;
    if (nama.trim() !== '') {
      const li = document.createElement('li');
      li.textContent = `✅ ${nama} hadir`;
      document.getElementById('daftarAbsensi').appendChild(li);
      document.getElementById('namaSiswa').value = '';
    } else {
      alert('Masukkan nama terlebih dahulu.');
    }
  }


  function tambahIde() {
    const ide = document.getElementById('ide').value;
    if (ide.trim() !== '') {
      const li = document.createElement('li');
      li.textContent = `💡 ${ide}`;
      document.getElementById('daftarIde').appendChild(li);
      document.getElementById('ide').value = '';
    } else {
      alert('Tulis ide kegiatan terlebih dahulu.');
    }
  }
</script>
